-- =============================================================================
-- 1. Enums
-- =============================================================================

-- Lifecycle state of a course. Drives what is visible to learners vs. authors.
--   draft     - being authored, not visible to learners
--   published - live and enrollable
--   archived  - hidden from catalog, existing enrollments remain valid
create type course_status as enum ('draft', 'published', 'archived');

-- Progress state of a single user's enrollment in a course.
--   active    - learner is currently working through the course
--   completed - learner has finished all required content
--   dropped   - learner has voluntarily left the course
create type enrollment_status as enum ('active', 'completed', 'dropped');

-- Lifecycle state of an agent memory record.
--   active   - memory is current and should be surfaced to the agent via MCP
--   stale    - memory is preserved but should not be used (e.g. user paused
--              the agent or memory is out of date and pending refresh)
--   archived - memory is retained for audit but excluded from MCP queries
create type agent_memory_status as enum ('active', 'stale', 'archived');

-- Lifecycle state of an Onti package subscription tier.
--   draft     - being authored, not purchasable
--   active    - available for purchase / assignment
--   retired   - hidden from new purchases, existing entitlements remain valid
create type onti_package_status as enum ('draft', 'active', 'retired');

-- Type of product a user_products row grants access to. Exactly one of the
-- product foreign key columns must be populated to match this kind.
create type product_kind as enum ('course', 'agent', 'onti_package');

-- Lifecycle of a user's access to a product.
--   active    - access is live
--   expired   - time-bounded access has lapsed
--   revoked   - access was explicitly removed (refund, abuse, etc.)
create type user_product_status as enum ('active', 'expired', 'revoked');


-- =============================================================================
-- 2. Tables
-- =============================================================================

-- -----------------------------------------------------------------------------
-- user_details
-- -----------------------------------------------------------------------------
-- Application-level profile data for an authenticated user.
--
-- Identity is owned by Supabase Auth (auth.users). We do NOT introduce a
-- separate surrogate id here — the primary key IS auth.users.id. This keeps
-- the relationship strictly 1:1 and avoids the confusion of two competing
-- user identifiers across the codebase.
--
-- A row should be created at signup (via trigger on auth.users or from the
-- backend immediately after signup) so user_details stays in lockstep with
-- auth.users.
-- -----------------------------------------------------------------------------
create table user_details (
  user_id          uuid primary key references auth.users(id) on delete cascade,
  display_name     text,
  avatar_url       text,
  date_of_birth    date,
  tags             text[] not null default '{}',
  last_active_at   timestamptz,
  created_at       timestamptz not null default now(),
  updated_at       timestamptz not null default now()
);

alter table user_details enable row level security;


-- -----------------------------------------------------------------------------
-- courses
-- -----------------------------------------------------------------------------
-- A Course is the core product offering. Users enroll in courses to consume
-- their content. Courses move through a lifecycle (draft → published →
-- archived) tracked by the course_status enum.
--
-- Read access: published courses are readable by any authenticated user.
-- Write access: managed by the backend via service_role (no direct learner
-- writes).
-- -----------------------------------------------------------------------------
create table courses (
  id          uuid primary key default gen_random_uuid(),
  title       text not null,
  description text,
  status      course_status not null default 'draft',
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now()
);

alter table courses enable row level security;


-- -----------------------------------------------------------------------------
-- course_enrollments
-- -----------------------------------------------------------------------------
-- Join table linking a user (auth.users.id) to a course they have enrolled
-- in. One row per (user, course) pair — the unique constraint prevents
-- duplicate enrollments.
--
-- enrolled_at  - when the user joined the course
-- completed_at - set when status transitions to 'completed'
--
-- Read access: a user can read only their own enrollments.
-- Write access: managed by the backend; learners cannot directly mutate
-- enrollment status (e.g. mark themselves complete) without going through
-- application logic.
-- -----------------------------------------------------------------------------
create table course_enrollments (
  id           uuid primary key default gen_random_uuid(),
  user_id      uuid not null references auth.users(id) on delete cascade,
  course_id    uuid not null references courses(id) on delete cascade,
  status       enrollment_status not null default 'active',
  enrolled_at  timestamptz not null default now(),
  completed_at timestamptz,
  created_at   timestamptz not null default now(),
  updated_at   timestamptz not null default now(),
  unique (user_id, course_id)
);

alter table course_enrollments enable row level security;


-- -----------------------------------------------------------------------------
-- course_onboardings
-- -----------------------------------------------------------------------------
-- Per-course onboarding submission from a learner. Each course can ship its
-- own onboarding questionnaire (intake form, preferences, prior knowledge,
-- etc.), and learners answer it when they enroll.
--
-- onboarding_version - integer version of the onboarding form the answers
--                      correspond to. Increment this in the application
--                      whenever the questionnaire shape changes so older
--                      submissions remain interpretable.
-- answers            - JSONB blob containing the learner's responses. The
--                      shape is owned by the application layer; do NOT
--                      branch SQL on its internal keys.
--
-- One submission per (user, course) pair. If the questionnaire is updated
-- and the learner re-submits, the row is updated in place and
-- onboarding_version is bumped accordingly.
-- -----------------------------------------------------------------------------
create table course_onboardings (
  id                 uuid primary key default gen_random_uuid(),
  user_id            uuid not null references auth.users(id) on delete cascade,
  course_id          uuid not null references courses(id) on delete cascade,
  onboarding_version integer not null,
  answers            jsonb not null default '{}'::jsonb,
  created_at         timestamptz not null default now(),
  updated_at         timestamptz not null default now(),
  unique (user_id, course_id)
);

alter table course_onboardings enable row level security;


-- -----------------------------------------------------------------------------
-- agents
-- -----------------------------------------------------------------------------
-- An Agent is our abstraction of an AI coach that users can interact with.
-- Each agent record points at a concrete underlying model hosted by a
-- provider (OpenAI, Anthropic, etc.). Access to a specific agent by a user
-- is gated by application logic (entitlements, plan, course enrollment,
-- etc.) and not modeled directly on this table.
--
-- provider   - the upstream model host (e.g. 'openai', 'anthropic').
-- model      - the model identifier as known to the provider
--              (e.g. 'gpt-4o', 'claude-sonnet-4-5'). Stored as text to
--              avoid coupling our schema to any one provider's catalog.
-- provider_agent_id - the id this agent is known by at the provider, if the
--              provider exposes long-lived agent/assistant objects.
--              Optional — null for stateless model-only agents.
-- -----------------------------------------------------------------------------
create table agents (
  id                uuid primary key default gen_random_uuid(),
  name              text not null,
  description       text,
  provider          text not null,
  model             text not null,
  provider_agent_id text,
  created_at        timestamptz not null default now(),
  updated_at        timestamptz not null default now()
);

alter table agents enable row level security;


-- -----------------------------------------------------------------------------
-- agent_memories
-- -----------------------------------------------------------------------------
-- Per-(user, agent) memory record. Stores long-lived conversational context
-- that an MCP server will later read so the agent can recall prior
-- interactions across sessions.
--
-- One row per (user, agent) pair — the unique constraint enforces that
-- memory is consolidated per relationship rather than appended per
-- conversation. Conversation-level history lives elsewhere; this table is
-- the rolling, summarized memory.
--
-- memory            - free-form natural-language summary the agent reads.
-- data              - JSONB structured memory (facts, preferences, goals,
--                     etc.). Shape is owned by the application; do NOT
--                     branch SQL on its internal keys.
-- last_updated_at   - when the memory was last meaningfully refreshed.
--                     Distinct from updated_at, which bumps on any row
--                     change (status flips, metadata edits, etc.).
-- status            - lifecycle of this memory (see agent_memory_status).
-- -----------------------------------------------------------------------------
create table agent_memories (
  id              uuid primary key default gen_random_uuid(),
  user_id         uuid not null references auth.users(id) on delete cascade,
  agent_id        uuid not null references agents(id) on delete cascade,
  memory          text not null default '',
  data            jsonb not null default '{}'::jsonb,
  status          agent_memory_status not null default 'active',
  last_updated_at timestamptz not null default now(),
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now(),
  unique (user_id, agent_id)
);

alter table agent_memories enable row level security;


-- -----------------------------------------------------------------------------
-- onti_packages
-- -----------------------------------------------------------------------------
-- An Onti package is a monthly subscription tier that grants special access
-- beyond any single course or agent (bundles, premium features, priority
-- support, etc.). Packages are themselves a product type that can be
-- granted to a user via user_products.
--
-- The exact entitlements a package confers are resolved by the backend at
-- access-check time; this table just identifies the tier itself.
-- -----------------------------------------------------------------------------
create table onti_packages (
  id                  uuid primary key default gen_random_uuid(),
  name                text not null,
  description         text,
  monthly_price_cents integer not null,
  currency            text not null default 'USD',
  status              onti_package_status not null default 'draft',
  created_at          timestamptz not null default now(),
  updated_at          timestamptz not null default now()
);

alter table onti_packages enable row level security;


-- -----------------------------------------------------------------------------
-- user_products
-- -----------------------------------------------------------------------------
-- The single source of truth for "what does this user have access to?".
-- A dedicated access service reads from this table to gate every product
-- in the system.
--
-- Each row links one user to exactly ONE product. The product is one of:
--   - a course        (course_id    is set, others null)
--   - an agent        (agent_id     is set, others null)
--   - an onti_package (package_id   is set, others null)
--
-- This is enforced by the check constraint below: exactly one product
-- foreign key must be non-null, and it must match the product_kind enum
-- column. Mixing product types on a single row would make access checks
-- ambiguous, so we explicitly forbid it.
--
-- One row per (user, product) pair. To grant a user multiple products,
-- insert multiple rows. To remove access, set status to 'revoked' or
-- 'expired' rather than deleting, so we retain history.
--
-- expires_at - optional end of access window. Null means no expiration
--              (typical for one-time course purchases). Subscription-style
--              access (e.g. onti_packages) sets this to the end of the
--              current billing period.
-- -----------------------------------------------------------------------------
create table user_products (
  id           uuid primary key default gen_random_uuid(),
  user_id      uuid not null references auth.users(id) on delete cascade,
  product_kind product_kind not null,
  course_id    uuid references courses(id) on delete cascade,
  agent_id     uuid references agents(id) on delete cascade,
  package_id   uuid references onti_packages(id) on delete cascade,
  status       user_product_status not null default 'active',
  granted_at   timestamptz not null default now(),
  expires_at   timestamptz,
  created_at   timestamptz not null default now(),
  updated_at   timestamptz not null default now(),

  -- Exactly one product reference must be set, and it must match product_kind.
  constraint user_products_exactly_one_product check (
    (product_kind = 'course'       and course_id  is not null and agent_id is null     and package_id is null) or
    (product_kind = 'agent'        and agent_id   is not null and course_id is null    and package_id is null) or
    (product_kind = 'onti_package' and package_id is not null and course_id is null    and agent_id is null)
  ),

  -- One row per (user, product) pair, per product type.
  constraint user_products_unique_course  unique (user_id, course_id),
  constraint user_products_unique_agent   unique (user_id, agent_id),
  constraint user_products_unique_package unique (user_id, package_id)
);

alter table user_products enable row level security;

create index courses_status_idx on courses (status);

create index course_onboardings_user_id_idx on course_onboardings (user_id);

-- Fast lookup of all submissions for a given course.
create index course_onboardings_course_id_idx on course_onboardings (course_id);

-- Fast lookup of a user's agent memories (MCP read path).
create index agent_memories_user_id_idx on agent_memories (user_id);

-- Fast lookup of all memories for a given agent.
create index agent_memories_agent_id_idx on agent_memories (agent_id);

-- Filter active memories quickly (MCP typically only wants 'active').
create index agent_memories_status_idx on agent_memories (status);

-- Hot path for the access service: "what does this user have?"
create index user_products_user_id_idx on user_products (user_id);

-- Reverse lookups: "who has this product?"
create index user_products_course_id_idx  on user_products (course_id)  where course_id  is not null;
create index user_products_agent_id_idx   on user_products (agent_id)   where agent_id   is not null;
create index user_products_package_id_idx on user_products (package_id) where package_id is not null;

-- Filter by product kind and status for entitlement queries.
create index user_products_kind_status_idx on user_products (product_kind, status);

-- Catalog listing of active packages.
create index onti_packages_status_idx on onti_packages (status);

-- ----- access-control helpers -------------------------------------------------
-- has_course_access(course_id) is the single source of truth for "can the
-- current user see this course's content?" It underlies RLS on courses and on
-- every table in the course-content hierarchy (modules, variants, contents,
-- assessment flows). A dropped enrollment revokes access; active/completed
-- enrollments retain it, including for archived courses (see course_status).
create or replace function has_course_access(p_course_id uuid)
returns boolean
language sql
stable
set search_path = public
as $$
  select exists (
    select 1
    from course_enrollments ce
    where ce.course_id = p_course_id
      and ce.user_id = auth.uid()
      and ce.status <> 'dropped'
  );
$$;

-- ----- user_details ----------------------------------------------------------
-- A user can read and update only their own profile row.

create policy "user_details: self read"
  on user_details for select
  using (auth.uid() = user_id);

create policy "user_details: self update"
  on user_details for update
  using (auth.uid() = user_id);


-- ----- courses ---------------------------------------------------------------
-- Any authenticated user can read published courses. Drafts and archived
-- courses are hidden from learners and managed via the backend.

create policy "courses: read published or enrolled"
  on courses for select
  to authenticated
  using (status = 'published' or has_course_access(id));


-- ----- course_enrollments ----------------------------------------------------
-- A learner can only see their own enrollments. Inserts and updates flow
-- through the backend (service_role), which bypasses RLS.

create policy "course_enrollments: self read"
  on course_enrollments for select
  using (auth.uid() = user_id);


-- ----- course_onboardings ----------------------------------------------------
-- A learner can read and submit/update their own onboarding answers.
-- Backend (service_role) handles cross-user reads (e.g. for instructors).

create policy "course_onboardings: self read"
  on course_onboardings for select
  using (auth.uid() = user_id);

create policy "course_onboardings: self insert"
  on course_onboardings for insert
  with check (auth.uid() = user_id);

create policy "course_onboardings: self update"
  on course_onboardings for update
  using (auth.uid() = user_id);


-- ----- agents ----------------------------------------------------------------
-- The agent catalog is readable by any authenticated user. Per-user
-- entitlement (which agents a given user can actually invoke) is enforced
-- by the backend, not at the SQL layer.

create policy "agents: read all"
  on agents for select
  to authenticated
  using (true);


-- ----- agent_memories --------------------------------------------------------
-- A user can read their own memories. Writes flow through the backend /
-- MCP server (service_role) so summarization and structure stay consistent.

create policy "agent_memories: self read"
  on agent_memories for select
  using (auth.uid() = user_id);


-- ----- onti_packages ---------------------------------------------------------
-- Active packages are visible to any authenticated user for catalog display.
-- Draft and retired packages are managed via the backend.

create policy "onti_packages: read active"
  on onti_packages for select
  to authenticated
  using (status = 'active');


-- ----- user_products ---------------------------------------------------------
-- A user can read their own entitlements (the access service uses
-- service_role to read across users). Writes are backend-only — users
-- cannot grant themselves access.

create policy "user_products: self read"
  on user_products for select
  using (auth.uid() = user_id);

-- -----------------------------------------------------------------------------
-- set_timestamps()
-- -----------------------------------------------------------------------------
-- Shared trigger function used by every table that has created_at and
-- updated_at columns.
--
-- On INSERT: forces created_at and updated_at to now(), so application code
--            cannot accidentally (or intentionally) set them to stale values.
-- On UPDATE: bumps updated_at to now() and preserves the original created_at.
--
-- This guarantees both timestamps are always server-authoritative.
-- -----------------------------------------------------------------------------
create or replace function set_timestamps()
returns trigger
language plpgsql
as $$
begin
  if tg_op = 'INSERT' then
    new.created_at = now();
    new.updated_at = now();
  elsif tg_op = 'UPDATE' then
    new.created_at = old.created_at;
    new.updated_at = now();
  end if;
  return new;
end;
$$;

-- Attach the trigger to every table with created_at / updated_at columns.

create trigger user_details_set_timestamps
  before insert or update on user_details
  for each row execute function set_timestamps();

create trigger courses_set_timestamps
  before insert or update on courses
  for each row execute function set_timestamps();

create trigger course_enrollments_set_timestamps
  before insert or update on course_enrollments
  for each row execute function set_timestamps();

create trigger course_onboardings_set_timestamps
  before insert or update on course_onboardings
  for each row execute function set_timestamps();

create trigger agents_set_timestamps
  before insert or update on agents
  for each row execute function set_timestamps();

create trigger agent_memories_set_timestamps
  before insert or update on agent_memories
  for each row execute function set_timestamps();

create trigger onti_packages_set_timestamps
  before insert or update on onti_packages
  for each row execute function set_timestamps();

create trigger user_products_set_timestamps
  before insert or update on user_products
  for each row execute function set_timestamps();


-- =============================================================================
-- 6. Billing (vendor-agnostic)
-- -----------------------------------------------------------------------------
-- Billing is modelled as a small, vendor-agnostic layer so we can swap or add
-- payment providers without changing application code. All provider-specific
-- identifiers live in `vendor_*_id` columns and are scoped by the `vendor`
-- enum below. Stripe is the only vendor today; new vendors (Paddle, Lemon
-- Squeezy, etc.) are added by extending the enum, never by adding columns.
--
-- Tables:
--   - billing_products          (catalog item from a vendor)
--   - billing_prices            (a price/SKU for a product)
--   - billing_checkout_sessions (a hosted checkout attempt)
--   - billing_events            (raw webhook events received from a vendor)
--
-- Webhook handling pattern: every inbound event is first persisted verbatim
-- into `billing_events` (idempotent on vendor_event_id). A separate processor
-- reads from that table and projects state into the other billing tables and
-- into `user_products`. This keeps ingestion cheap, makes replay trivial, and
-- gives us a complete audit log of everything the vendor told us.
-- =============================================================================


-- ----- enums -----------------------------------------------------------------

-- Payment vendor that owns a given billing record. Add new variants here when
-- onboarding a new processor; never reuse a variant for a different vendor.
create type billing_vendor as enum ('stripe');

-- Lifecycle of a vendor product mirrored into our catalog.
--   active   - sellable / referenced by active prices
--   archived - no longer sold, retained for historical records
create type billing_product_status as enum ('active', 'archived');

-- Lifecycle of a vendor price.
--   active   - currently sellable
--   archived - no longer sellable, retained for historical records
create type billing_price_status as enum ('active', 'archived');

-- Billing model for a price.
--   one_time  - single charge (e.g. a course purchase)
--   recurring - subscription-style charge (e.g. onti_packages)
create type billing_price_type as enum ('one_time', 'recurring');

-- Recurrence cadence for a recurring price. Null for one_time prices.
create type billing_price_interval as enum ('day', 'week', 'month', 'year');

-- Lifecycle of a hosted checkout session, mirrored from the vendor.
--   open      - session created, awaiting payment
--   complete  - payment captured (or async payment succeeded later)
--   expired   - session timed out without payment
--   failed    - async payment attempt failed
create type billing_checkout_status as enum ('open', 'complete', 'expired', 'failed');

-- Stripe webhook event types we currently subscribe to. Listed explicitly so
-- the enum doubles as documentation of the contract with the vendor. Add new
-- variants here before subscribing to additional events.
create type billing_event_type as enum (
  'checkout.session.async_payment_failed',
  'checkout.session.async_payment_succeeded',
  'checkout.session.completed',
  'checkout.session.expired',
  'invoice.created',
  'invoice.deleted',
  'invoice.paid',
  'invoice.payment_succeeded',
  'refund.created',
  'refund.failed',
  'refund.updated'
);

-- Processing state for a received webhook event.
--   pending   - persisted, not yet processed by the projector
--   processed - successfully applied to downstream tables
--   failed    - projector errored; see error_message and retry
--   ignored   - intentionally not acted upon (e.g. duplicate / out of order)
create type billing_event_status as enum ('pending', 'processed', 'failed', 'ignored');


-- ----- billing_products ------------------------------------------------------
-- A product as defined in the payment vendor's catalog. One row per
-- (vendor, vendor_product_id). We keep just enough data to render a checkout
-- and to link a purchase back to something we sell internally; the vendor
-- remains the source of truth for everything else.
--
-- onti_product_kind / onti_product_id are an optional mapping back to our
-- own catalog (courses, agents, onti_packages). They are null until a human
-- (or admin tool) links the vendor product to the thing it grants.

create table billing_products (
  id                uuid primary key default gen_random_uuid(),
  vendor            billing_vendor not null,
  vendor_product_id text not null,
  name              text not null,
  description       text,
  status            billing_product_status not null default 'active',
  onti_product_kind product_kind,
  onti_product_id   uuid,
  metadata          jsonb not null default '{}'::jsonb,
  created_at        timestamptz not null default now(),
  updated_at        timestamptz not null default now(),

  constraint billing_products_vendor_unique unique (vendor, vendor_product_id),

  -- If onti_product_kind is set, onti_product_id must be set, and vice versa.
  constraint billing_products_onti_mapping_complete check (
    (onti_product_kind is null and onti_product_id is null) or
    (onti_product_kind is not null and onti_product_id is not null)
  )
);

alter table billing_products enable row level security;


-- ----- billing_prices --------------------------------------------------------
-- A specific price/SKU attached to a billing_product. A product can have many
-- prices (e.g. monthly vs. yearly, USD vs. EUR). Amounts are stored in the
-- smallest currency unit (cents) to match every major vendor's API.
--
-- For recurring prices, interval and interval_count together describe the
-- cadence (e.g. interval='month', interval_count=1 -> monthly). Both are null
-- for one_time prices, enforced by the check constraint below.

create table billing_prices (
  id                uuid primary key default gen_random_uuid(),
  vendor            billing_vendor not null,
  vendor_price_id   text not null,
  product_id        uuid not null references billing_products(id) on delete cascade,
  type              billing_price_type not null,
  unit_amount       integer not null,
  currency          text not null,
  interval          billing_price_interval,
  interval_count    integer,
  status            billing_price_status not null default 'active',
  metadata          jsonb not null default '{}'::jsonb,
  created_at        timestamptz not null default now(),
  updated_at        timestamptz not null default now(),

  constraint billing_prices_vendor_unique unique (vendor, vendor_price_id),

  -- Recurring prices must have a cadence; one_time prices must not.
  constraint billing_prices_recurrence_matches_type check (
    (type = 'one_time'  and interval is null     and interval_count is null) or
    (type = 'recurring' and interval is not null and interval_count is not null and interval_count > 0)
  )
);

alter table billing_prices enable row level security;


-- ----- billing_checkout_sessions ---------------------------------------------
-- A hosted checkout attempt initiated for a user. We create the row when we
-- ask the vendor to create the session, then update status as webhook events
-- come in. user_id is nullable because some sessions (e.g. anonymous gift
-- purchases) may not be tied to a logged-in user at creation time.
--
-- price_id is the price the user was checking out against at the time the
-- session was created. We snapshot it here so price edits later don't rewrite
-- history.

create table billing_checkout_sessions (
  id                          uuid primary key default gen_random_uuid(),
  vendor                      billing_vendor not null,
  vendor_checkout_session_id  text not null,

-- 7. Admin dashboard access
-- -----------------------------------------------------------------------------
-- Grants a subset of users access to the internal admin dashboard and stamps
-- their role into the Supabase Auth JWT so the dashboard (and downstream RLS,
-- if we ever need it) can branch on `auth.jwt() -> 'admin_dashboard_role'`.
--
-- This block is self-contained: it can be executed independently of the rest
-- of the schema as long as auth.users already exists. Drop the objects in
-- reverse order if you need to roll it back.
--
-- Design notes:
--   - Most users will NEVER have a row in admin_dashboard_access. Absence of
--     a row == no admin access. We don't store a "none" role.
--   - We claim into the JWT via a Supabase Auth `custom_access_token_hook`.
--     The hook fires on every token issuance (sign-in and refresh), so role
--     changes propagate within one refresh cycle without touching auth.users.
--   - The hook MUST be wired up in Supabase Auth settings:
--       Authentication → Hooks → Custom Access Token → public.admin_dashboard_jwt_hook
-- =============================================================================


-- ----- enums -----------------------------------------------------------------

-- Role inside the admin dashboard. Only users with a row in
-- admin_dashboard_access carry one of these; everyone else has no admin role
-- at all (their JWT simply omits the claim).
--   admin   - full read/write across the dashboard
--   analyst - read-only access to analytics and reporting surfaces
create type admin_dashboard_role as enum ('admin', 'analyst');


-- ----- admin_dashboard_access ------------------------------------------------
-- One row per user that has any admin dashboard permission. The primary key
-- IS auth.users.id, so the relationship is strictly 1:1 — a user either has
-- admin access (with exactly one role) or they don't.
--
-- granted_by points at the admin who provisioned this access, for audit.
-- Null is allowed for the initial bootstrap admin who has no grantor.

create table admin_dashboard_access (
  user_id     uuid primary key references auth.users(id) on delete cascade,
  role        admin_dashboard_role not null,
  granted_by  uuid references auth.users(id) on delete set null,
  granted_at  timestamptz not null default now(),
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now()
);

alter table admin_dashboard_access enable row level security;

-- Reverse lookup: "who are all the admins / analysts?"
create index admin_dashboard_access_role_idx on admin_dashboard_access (role);

-- A user can see their own admin row (so the dashboard can confirm their role
-- client-side). All writes are backend-only via service_role.
create policy "admin_dashboard_access: self read"
  on admin_dashboard_access for select
  using (auth.uid() = user_id);

create trigger admin_dashboard_access_set_timestamps
  before insert or update on admin_dashboard_access
  for each row execute function set_timestamps();


-- ----- admin_dashboard_jwt_hook ----------------------------------------------
-- Supabase Auth custom access token hook. On every token issuance, Supabase
-- calls this function with the event payload, we look up the user's admin
-- access (if any), and merge an `admin_dashboard_role` claim into the JWT
-- under the standard `claims` object.
--
-- Contract (Supabase):
--   - Input  jsonb: { user_id, claims, authentication_method, ... }
--   - Output jsonb: same shape, with `claims` possibly augmented.
--   - Must be SECURITY DEFINER so it can read admin_dashboard_access while
--     RLS is on, and must be owned by a role with select on that table.
--
-- Behavior:
--   - If the user has a row in admin_dashboard_access, set
--       claims.admin_dashboard_role = '<role>'
--     so the dashboard frontend can read it from the JWT directly without an
--     extra round-trip.
--   - If the user has no row (the common case for the vast majority of users),
--     return the event unchanged. We do NOT set the claim to null / 'none' —
--     absence of the claim is the signal that the user has no admin access.

create or replace function admin_dashboard_jwt_hook(event jsonb)
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  v_user_id uuid;
  v_role    admin_dashboard_role;
  v_claims  jsonb;
begin
  v_user_id := (event ->> 'user_id')::uuid;
  v_claims  := coalesce(event -> 'claims', '{}'::jsonb);

  select role
    into v_role
    from admin_dashboard_access
   where user_id = v_user_id;

  if v_role is not null then
    v_claims := jsonb_set(v_claims, '{admin_dashboard_role}', to_jsonb(v_role::text), true);
    event    := jsonb_set(event, '{claims}', v_claims, true);
  end if;

  return event;
end;
$$;

-- Lock the function down: only the Supabase Auth admin role (which invokes
-- the hook) and service_role (for testing / ops) should be able to call it.
revoke execute on function admin_dashboard_jwt_hook(jsonb) from public, anon, authenticated;
grant  execute on function admin_dashboard_jwt_hook(jsonb) to supabase_auth_admin, service_role;

-- The hook needs to read admin_dashboard_access at token-issuance time.
grant select on admin_dashboard_access to supabase_auth_admin;

-- 8. Product access
-- -----------------------------------------------------------------------------
-- Tracks every attempt by a buyer to obtain access to a billing product, from
-- the moment a checkout session is created through to expiration. This is the
-- table the "do they have access?" feature reads from for *purchases in
-- flight*, and the durable record that ties a billing flow to the thing it
-- grants (a course today; agents / onti_packages later via the same shape).
--
-- Relationship to other tables:
--   - billing_checkout_sessions → product_access.checkout_session_id (1:1 per
--     attempt; one row in product_access is created when the checkout session
--     is created).
--   - billing_events            → product_access transitions are *driven* by
--     incoming events. The webhook projector reads `billing_events` and
--     updates the matching `product_access` row by `checkout_session_id`.
--   - billing_products          → product_access.billing_product_id pins the
--     vendor product being purchased.
--   - course_enrollments        → product_access.course_enrollment_id is set
--     once the buyer is reconciled and the enrollment is provisioned.
--   - auth.users                → product_access.user_id is OPTIONAL: many
--     buyers pay before their app account exists (anonymous checkout from a
--     landing page). `email` is captured up-front and the projector / signup
--     trigger links the row back to `user_id` once the account is created.
--
-- One buyer can have many product_access rows — one per product attempt. They
-- are not collapsed when a user re-buys; history is preserved and the access
-- service reads the most recent row with an active-equivalent status.
--
-- Note: `user_products` remains the source of truth for *granted, currently
-- valid* entitlements. `product_access` is the lifecycle log of a purchase
-- and is what the dashboard and reconciliation jobs read. The webhook
-- projector writes to both when a checkout completes.
-- =============================================================================


-- ----- enums -----------------------------------------------------------------

-- Lifecycle of a product_access row.
--   checkout              - checkout session created at the vendor, awaiting
--                           buyer payment. Row inserted by the checkout
--                           endpoint.
--   pending_confirmation  - vendor reported the payment is being processed
--                           asynchronously (e.g. SEPA debit, bank transfer).
--                           No access granted yet.
--   paid                  - one-time purchase: payment captured, access is
--                           live. Terminal for one_time products until
--                           expiration.
--   expired               - access window has lapsed (subscription cancelled
--                           and period ended, or one-time access had an
--                           explicit expires_at that passed).
--   subscribed            - recurring purchase: subscription is active and in
--                           good standing. Distinguishes "currently paying
--                           every period" from a one-shot `paid`.
create type product_access_status as enum (
  'checkout',
  'pending_confirmation',
  'paid',
  'expired',
  'subscribed'
);

-- Billing model the access row was created under. Mirrors billing_prices.type
-- but is snapshotted here so the lifecycle can be reasoned about without a
-- join back to the price (which may itself be archived).
--   one_time     - single charge, terminal state is `paid` → eventually
--                  `expired` if there is an expires_at.
--   subscription - recurring charge, terminal state is `subscribed` → flips
--                  to `expired` when the subscription ends.
create type product_access_type as enum ('one_time', 'subscription');


-- ----- product_access --------------------------------------------------------

create table product_access (
  id                     uuid primary key default gen_random_uuid(),

  -- Buyer. user_id is optional: a checkout from a landing page may complete
  -- before the buyer creates their app account. email is captured at checkout
  -- and used to reconcile to auth.users once the account exists.
  user_id                uuid references auth.users(id) on delete set null,
  email                  text not null,

  -- What was bought. billing_product_id is required; course_id is set only
  -- when the underlying billing_product maps to a course (resolved at
  -- checkout time via billing_products.onti_product_kind / onti_product_id).
  billing_product_id     uuid not null references billing_products(id) on delete restrict,
  course_id              uuid references courses(id) on delete set null,

  -- The flow that produced this row. One product_access per checkout session.
  checkout_session_id    uuid not null references billing_checkout_sessions(id) on delete cascade,

  -- Once the buyer is reconciled and a course enrollment is provisioned, we
  -- pin the resulting enrollment here so the access service can join cleanly.
  course_enrollment_id   uuid references course_enrollments(id) on delete set null,

  -- Lifecycle.
  status                 product_access_status not null default 'checkout',
  type                   product_access_type not null,

  -- Money snapshot. We do NOT re-read billing_prices later — these reflect
  -- what the buyer actually paid as confirmed by the vendor webhook.
  amount_paid            integer,
  currency               text,

  -- Relevant timestamps along the lifecycle.
  checkout_started_at    timestamptz not null default now(),
  confirmed_at           timestamptz,
  paid_at                timestamptz,
  access_starts_at       timestamptz,
  access_expires_at      timestamptz,

  created_at             timestamptz not null default now(),
  updated_at             timestamptz not null default now(),

  -- One row per checkout session. Guarantees the webhook projector can
  -- upsert by checkout_session_id without ambiguity.
  constraint product_access_checkout_session_unique unique (checkout_session_id),

  -- If course_enrollment_id is set, course_id must be set too (the

  constraint product_access_enrollment_implies_course check (

);

-- 9. Course content structure
-- -----------------------------------------------------------------------------
-- Models the internal structure of a course: modules (ordered sections) →
-- variants (personalised paths within a module) → contents (atomic items).
-- Each enrolled learner is assigned exactly one variant per module, giving
-- every alumnus a unique content sequence. Progress and answers are tracked
-- at the content level.
--
-- Personalisation operates at the MODULE level, never the course level: the
-- set of modules in a course, and their order, is fixed and identical for
-- every learner. What varies per learner is only which variant (content
-- sequence) they get within each module — see course_modules vs.
-- course_variants below.
--
-- Hierarchy:
--   courses  (existing)
--     └── course_modules       (ordered sections of a course)
--           └── course_variants (one or more paths through a module)
--                 └── course_contents (ordered atomic content items)
--
-- Per-learner state:
--   course_alumni_assignments  maps (user, module) → variant
--   student_progress           records completion of each content item
--   student_answers            captures answers for content that requires them
-- =============================================================================


-- ----- enums -----------------------------------------------------------------

-- Discriminates the kind of content a course_contents row represents.
-- url / body usage per type:
--   html       - body holds the markup; url is null or links a referenced asset
--   video      - url points to the CDN/streaming asset; body is null
--   audio      - url points to the audio file; body is null
--   meditation - url points to the guided audio/video file; body is null
--   quiz       - body holds the question JSON schema; url is null
create type course_content_type as enum ('html', 'video', 'audio', 'meditation', 'quiz');


-- ----- course_modules --------------------------------------------------------
-- An ordered section (chapter) of a course. The set of modules and their
-- order is FIXED per course — identical for every learner. Personalisation
-- never adds, removes, or reorders modules; each module offers one or more
-- variants (see course_variants) so the backend can personalise the path
-- *within* that module only.
--
-- position - ascending sort order within the parent course. The application
--            is responsible for keeping positions contiguous.

create table course_modules (
  id          uuid primary key default gen_random_uuid(),
  course_id   uuid not null references courses(id) on delete cascade,
  title       text not null,
  description text,
  position    integer not null default 0,
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now()
);

alter table course_modules enable row level security;


-- ----- course_variants -------------------------------------------------------
-- A named path through a module — an ordered set of content items. A module
-- must have at least one variant; multiple variants allow the system to assign
-- different experiences to different learners (A/B or fully personalised).
--
-- Scope: a variant belongs to exactly one module and only affects that
-- module's content sequence. There is no course-wide "path" — a learner's
-- variant in one module has no bearing on their variant in another; each
-- module's assignment is resolved independently (see
-- course_alumni_assignments).

create table course_variants (
  id          uuid primary key default gen_random_uuid(),
  module_id   uuid not null references course_modules(id) on delete cascade,
  name        text not null,
  description text,
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now()
);

alter table course_variants enable row level security;


-- ----- course_contents -------------------------------------------------------
-- An atomic learning item inside a variant, ordered by `position` (ascending).
-- `type` controls how the frontend renders the item and which storage field
-- is populated:
--   html / quiz   → body holds the content; url is typically null.
--   video / audio
--   / meditation  → url points to the media asset; body is null.

create table course_contents (
  id         uuid primary key default gen_random_uuid(),
  variant_id uuid not null references course_variants(id) on delete cascade,
  type       course_content_type not null,
  title      text not null,
  url        text,
  body       text,
  position   integer not null default 0,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table course_contents enable row level security;


-- ----- course_alumni_assignments ---------------------------------------------
-- Records which variant a given learner has been assigned for each module.
-- The unique constraint on (user_id, module_id) guarantees exactly one variant
-- per learner per module. The backend assigns variants at enrollment time or
-- lazily as the learner progresses; the application must ensure variant_id
-- references a variant that belongs to the specified module_id.

create table course_alumni_assignments (
  id          uuid primary key default gen_random_uuid(),
  user_id     uuid not null references auth.users(id) on delete cascade,
  module_id   uuid not null references course_modules(id) on delete cascade,

  assigned_at timestamptz not null default now(),
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now(),

  unique (user_id, module_id)
);

alter table course_alumni_assignments enable row level security;


-- ----- student_progress ------------------------------------------------------
-- One row per (learner, content item). Inserted when a learner opens a piece

create table student_progress (

-- ----- student_answers -------------------------------------------------------
-- Captures a learner's answers for content items that require a response
-- (type = 'quiz' or any other answer-bearing type). `answers` is a JSONB blob
-- whose shape is owned by the application; the SQL layer does not inspect its
-- keys. One submission per (learner, content) — re-submissions update the row
-- in place, preserving the latest answers.

create table student_answers (
  id           uuid primary key default gen_random_uuid(),
  user_id      uuid not null references auth.users(id) on delete cascade,

  submitted_at timestamptz not null default now(),

  updated_at   timestamptz not null default now(),

alter table student_answers enable row level security;


-- ----- student_insight_memory ------------------------------------------------
-- A timestamped insight or milestone marker generated during a learner's
-- interaction with a specific piece of content. Unlike student_progress (which
-- is a binary started/completed flag) this table captures *what* was notable:
-- a realisation, a struggle, a key question, or any signal the system or agent
-- deems worth remembering.
--
-- Multiple insights can be recorded for the same (user, content) pair — there

--
-- status reuses agent_memory_status (active / stale / archived) so the same
-- MCP query patterns that filter agent_memories work here unchanged.
-- insight  - free-form natural-language description the agent reads.
-- data     - JSONB structured metadata (tags, scores, context snippets, etc.).
--            Shape is owned by the application; do NOT branch SQL on its keys.

create table student_insight_memory (
  id         uuid primary key default gen_random_uuid(),
  user_id    uuid not null references auth.users(id) on delete cascade,
  content_id uuid not null references course_contents(id) on delete cascade,
  insight    text not null,
  data       jsonb not null default '{}'::jsonb,
  status     agent_memory_status not null default 'active',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table student_insight_memory enable row level security;

-- ----- RLS policies ----------------------------------------------------------

-- Course structure and content are gated behind actual course access — not
-- merely "authenticated" — since course_contents carries the paid material
-- (video/audio urls, quiz bodies) that the billing/enrollment model exists to
-- protect. Writes are backend-only via service_role.

-- has_module_access(module_id) extends has_course_access() one level down the
-- hierarchy so course_variants and downstream policies can check access via
-- the module's parent course without re-deriving the join each time.
create or replace function has_module_access(p_module_id uuid)
returns boolean
language sql
stable
set search_path = public
as $$
  select exists (
    select 1
    from course_modules cm
    where cm.id = p_module_id
      and has_course_access(cm.course_id)
  );
$$;

create policy "course_modules: read if enrolled"
  on course_modules for select
  to authenticated
  using (has_course_access(course_id));

create policy "course_variants: read if enrolled"
  on course_variants for select
  to authenticated
  using (has_module_access(module_id));

-- Contents are scoped to the learner's *assigned* variant, not merely "any
-- variant of a course they can access" — a learner has no need to see the
-- other personalised paths, and it keeps the leak surface minimal.
create policy "course_contents: read own assigned variant"
  on course_contents for select
  to authenticated
  using (
    exists (
      select 1
      from course_alumni_assignments caa
      join course_variants cv on cv.id = caa.variant_id

-- =============================================================================
-- 10. Assessment flows & variant assignment rules
-- -----------------------------------------------------------------------------
-- Enables structured quizzes or onboarding forms to run at configurable points
-- in a course (before the course, before a module, etc.) and automatically
-- assign the learner to the correct course variant based on their answers.
--
-- The existing course_alumni_assignments table remains the destination for all
-- variant assignments; this section provides the machinery to drive it.
--
-- Flow:
--   1. A variant_assignment_policy is defined for a course or module, pointing
--      at an assessment_flow and declaring when it fires (trigger_point).
--   2. The learner completes the flow → one student_assessment_responses row.
--   3. The backend evaluates variant_assignment_rules in priority order.
--   4. The first rule whose conditions match → variant is assigned via
--      course_alumni_assignments. default_variant_id is the fallback.
--
-- Scope: a policy — regardless of trigger_point — only ever assigns a variant
-- for ONE module: whichever module owns the course_variants row referenced by
-- the matched rule's target_variant_id (or the policy's default_variant_id).
-- trigger_point controls WHEN the assessment fires (before the course starts,
-- before/after a specific module, or standalone) — it does not widen the
-- assignment's scope. In particular, 'before_course' does NOT mean the whole
-- course's path is personalised; it only means this particular module's
-- assessment happens to run before the learner starts anything. Path/variant
-- selection is always per-module (see course_modules / course_variants). To
-- personalise more than one module from a shared signal, create one policy
-- (and rule set) per module — there is no mechanism for a single rule to
-- assign variants across multiple modules at once.
--
-- Conditions are stored as JSONB inside variant_assignment_rules (same
-- app-owns-shape convention as student_answers.answers). The SQL layer does
-- not introspect condition keys.
-- =============================================================================


-- ----- enums -----------------------------------------------------------------

-- Question types inside an assessment flow.
create type assessment_question_type as enum (
  'single_choice',
  'multiple_choice',
  'scale',
  'text',
  'boolean'
);

-- When in the learner's journey to trigger the assessment. This is timing
-- only, not scope — every trigger_point, including 'before_course', still
-- resolves to a single module's variant assignment (see the "Scope" note
-- above, in this section's header comment).
create type policy_trigger_point as enum (
  'before_course',
  'before_module',
  'after_module',
  'standalone'
);

-- Whether all conditions in a rule must pass (AND) or any one is enough (OR).
create type rule_match_mode as enum ('all', 'any');


-- ----- assessment_flows ------------------------------------------------------
-- A standalone questionnaire definition. Reusable across courses and modules —
-- the policy (below) is what binds a flow to a specific context.
--
-- version - increment in the application whenever the question set changes
--           shape, so existing student_assessment_responses remain interpretable.

create table assessment_flows (
  id          uuid primary key default gen_random_uuid(),
  title       text not null,
  description text,
  version     integer not null default 1,
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now()
);

alter table assessment_flows enable row level security;


-- ----- assessment_questions --------------------------------------------------
-- Ordered questions within a flow.
--
-- options  - JSONB array of {value, label} objects for choice types; null for
--            scale / text / boolean questions.
-- required - whether the learner must answer before submitting the flow.

create table assessment_questions (
  id         uuid primary key default gen_random_uuid(),
  flow_id    uuid not null references assessment_flows(id) on delete cascade,
  type       assessment_question_type not null,
  question   text not null,
  options    jsonb,
  position   integer not null default 0,
  required   boolean not null default true,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table assessment_questions enable row level security;


-- ----- student_assessment_responses ------------------------------------------
-- One row per (user, flow). Answers are stored as a JSONB object keyed by
-- question_id, mirroring the course_onboardings.answers pattern. Re-submissions
-- update the row in place; completed_at is stamped when the learner finishes.

create table student_assessment_responses (
  id           uuid primary key default gen_random_uuid(),
  user_id      uuid not null references auth.users(id) on delete cascade,
  flow_id      uuid not null references assessment_flows(id) on delete cascade,
  answers      jsonb not null default '{}'::jsonb,
  completed_at timestamptz,
  created_at   timestamptz not null default now(),
  updated_at   timestamptz not null default now(),

  unique (user_id, flow_id)
);

alter table student_assessment_responses enable row level security;


-- ----- variant_assignment_policies -------------------------------------------
-- Binds an assessment flow to a course or module context, declares when it
-- fires, and sets a default fallback variant for when no rule matches.
--
-- The check constraint ensures the trigger_point is coherent with the context
-- foreign keys: course-level triggers require course_id; module-level triggers
-- require module_id; standalone needs neither.
--
-- course_id on a 'before_course' policy only scopes WHEN the assessment fires
-- and which course's access gates it (see has_policy_access) — it does not
-- widen the assignment. default_variant_id (like target_variant_id below)
-- still points at one course_variants row belonging to one module, so even a
-- 'before_course' policy only ever assigns that single module's variant.

create table variant_assignment_policies (
  id                  uuid primary key default gen_random_uuid(),
  name                text not null,
  assessment_flow_id  uuid not null references assessment_flows(id) on delete restrict,
  course_id           uuid references courses(id) on delete cascade,
  module_id           uuid references course_modules(id) on delete cascade,
  trigger_point       policy_trigger_point not null,
  default_variant_id  uuid references course_variants(id) on delete set null,
  created_at          timestamptz not null default now(),
  updated_at          timestamptz not null default now(),

  constraint variant_assignment_policies_context_check check (
    (trigger_point = 'standalone' and course_id is null and module_id is null) or
    (trigger_point = 'before_course' and course_id is not null and module_id is null) or
    (trigger_point in ('before_module', 'after_module') and module_id is not null and course_id is null)
  )
);

alter table variant_assignment_policies enable row level security;


-- ----- variant_assignment_rules ----------------------------------------------
-- Priority-ordered rules within a policy. The backend evaluates rules in
-- ascending priority order; the first rule whose conditions pass determines
-- the variant assigned to the learner.
--
-- conditions - JSONB array of condition objects. Shape is owned by the
--              application; the SQL layer does not inspect keys. Each object
--              is expected to carry: question_id, operator, expected.
--              Supported operators: eq | neq | in | not_in | gte | lte.
-- match_mode - 'all' = every condition must pass (AND);
--              'any' = at least one condition must pass (OR).

create table variant_assignment_rules (
  id                uuid primary key default gen_random_uuid(),
  policy_id         uuid not null references variant_assignment_policies(id) on delete cascade,
  name              text not null,
  target_variant_id uuid not null references course_variants(id) on delete cascade,
  priority          integer not null default 0,
  match_mode        rule_match_mode not null default 'all',
  conditions        jsonb not null default '[]'::jsonb,
  created_at        timestamptz not null default now(),
  updated_at        timestamptz not null default now()
);

alter table variant_assignment_rules enable row level security;


-- ----- indexes ---------------------------------------------------------------

create index assessment_questions_flow_position_idx

-- "Has this user completed this flow?" (policy trigger check).

-- ----- RLS policies ----------------------------------------------------------

-- has_policy_access mirrors has_course_access/has_module_access one level up:
-- a policy is visible if it's standalone (not tied to any course/module — e.g.
-- a generic intake form) or if the current user has access to the course/
-- module it's scoped to.
create or replace function has_policy_access(
  p_trigger_point policy_trigger_point,
  p_course_id uuid,
  p_module_id uuid
)
returns boolean
language sql
stable
set search_path = public
as $$
  select
    p_trigger_point = 'standalone'
    or (p_course_id is not null and has_course_access(p_course_id))
    or (p_module_id is not null and has_module_access(p_module_id));
$$;

-- has_assessment_flow_access(flow_id) resolves through whichever policy binds
-- a flow to a course/module context.
create or replace function has_assessment_flow_access(p_flow_id uuid)
returns boolean
language sql
stable
set search_path = public
as $$
  select exists (
    select 1
    from variant_assignment_policies vap
    where vap.assessment_flow_id = p_flow_id
      and has_policy_access(vap.trigger_point, vap.course_id, vap.module_id)
  );
$$;

-- Assessment content is scoped to the same course/module access as the
-- content it gates — not merely "any authenticated user" — since flows can
-- reveal unpublished course structure or be tied to paid content.
create policy "assessment_flows: read if accessible"
  on assessment_flows for select
  to authenticated
  using (has_assessment_flow_access(id));

create policy "assessment_questions: read if accessible"
  on assessment_questions for select
  to authenticated
  using (has_assessment_flow_access(flow_id));

-- Learners can read and submit their own responses.

create policy "student_assessment_responses: self read"

  using (auth.uid() = user_id);

create policy "student_assessment_responses: self insert"
  on student_assessment_responses for insert
  with check (auth.uid() = user_id);

create policy "student_assessment_responses: self update"
  on student_assessment_responses for update
  using (auth.uid() = user_id)
  with check (auth.uid() = user_id);

-- Policies are readable once the learner has access to the course/module they
-- target (or if standalone). Rules remain backend-only — learners must not
-- see the assignment logic.

create policy "variant_assignment_policies: read if accessible"
  on variant_assignment_policies for select
  to authenticated
  using (has_policy_access(trigger_point, course_id, module_id));

-- =============================================================================
-- 11. WhatsApp messaging
-- -----------------------------------------------------------------------------
-- The WhatsApp *interface* layer: the thread with a contact and the append-only
-- log of every message sent or received through the WhatsApp Cloud API (Meta).
-- This is transport/audit state only — it does not model agent memory or LLM
-- context. Deciding what to reply, and remembering anything across messages,
-- is the job of agents / agent_memories (see section 2), which reads and
-- writes independently of these tables.
--
-- Tables:
--   - whatsapp_conversations (the continuous thread with one contact)
--   - whatsapp_messages      (append-only log of every inbound/outbound message)
--
-- MVP scope: phone-number-only. There is intentionally no auth.users link here
-- (no user_id column) — a contact is identified purely by wa_id. Reconciling a
-- WhatsApp contact to a known Onti account, if ever needed, can be added later
-- as an additional nullable column without changing this shape.
--
-- Webhook handling pattern mirrors billing_events (section 6): every inbound
-- message and status update is persisted directly (there is no separate raw
-- ingest + projector split here, since there is no multi-step downstream
-- projection to do — persisting *is* the whole job of the webhook handler).
-- =============================================================================


-- ----- enums -----------------------------------------------------------------

-- Who sent a given whatsapp_messages row.
create type whatsapp_message_direction as enum ('inbound', 'outbound');

-- Kind of WhatsApp message content. Mirrors the type values the WhatsApp Cloud
-- API itself uses; 'unknown' is a catch-all for types Meta adds later that we
-- haven't explicitly modeled yet (the full payload is always preserved in
-- whatsapp_messages.raw regardless of type).
create type whatsapp_message_type as enum (
  'text',
  'image',
  'video',
  'audio',
  'document',
  'sticker',
  'location',
  'contacts',
  'interactive',
  'button',
  'template',
  'order',
  'system',
  'unknown'
);

-- Lifecycle of a whatsapp_messages row.
--   received  - inbound message persisted from the webhook.
--   sent      - outbound message accepted by Meta.
--   delivered - Meta confirmed delivery to the recipient's device.
--   read      - recipient opened the message (read receipt).
--   failed    - outbound send was rejected by Meta, or delivery failed.
create type whatsapp_message_status as enum ('received', 'sent', 'delivered', 'read', 'failed');


-- ----- whatsapp_conversations --------------------------------------------------
-- One row per (business_phone_number_id, wa_id) pair — the continuous thread
-- with a contact, not a per-session log. WhatsApp itself has no notion of
-- discrete conversation boundaries beyond the 24-hour customer service window,
-- so this table tracks that window rather than a lifecycle status.
--
-- business_phone_number_id - Meta's phone_number_id for the WhatsApp Business
--                             number the thread is on. Stored per-row (rather
--                             than assumed to be a single global constant) so
--                             a second business number can be onboarded later
--                             without a schema change.
-- wa_id                     - the contact's WhatsApp id (phone number, digits
--                             only, no leading '+').
-- last_inbound_at           - timestamp of the contact's most recent message.
--                             Drives the 24h customer-service-window check
--                             that gates free-form (type = 'text') sends.
-- last_outbound_at          - timestamp of our most recent message to them.

create table whatsapp_conversations (
  id                       uuid primary key default gen_random_uuid(),
  business_phone_number_id text not null,
  wa_id                    text not null,
  contact_name             text,
  last_inbound_at          timestamptz,
  last_outbound_at         timestamptz,
  created_at               timestamptz not null default now(),
  updated_at               timestamptz not null default now(),

  constraint whatsapp_conversations_phone_wa_unique unique (business_phone_number_id, wa_id)
);

alter table whatsapp_conversations enable row level security;


-- ----- whatsapp_messages ------------------------------------------------------
-- Append-only log of every message sent or received on a conversation. Mirrors
-- billing_events' "store the raw payload verbatim" convention (see section 6)
-- so nothing Meta sends is ever lost, even for message types we don't fully
-- model in dedicated columns.
--
-- wamid        - Meta's message id. Nullable because a failed outbound send
--                never receives one from Meta, but the attempt is still
--                logged for audit. Unique when present (see index below) so
--                Meta's at-least-once webhook retries collapse into one row.
-- text_body    - populated for type = 'text' and for the selected option's
--                title on button/interactive replies; null otherwise.
-- context_wamid - the wamid of a quoted/replied-to message, if any.
-- error        - Meta's error object; populated only when status = 'failed'.
-- raw          - the full Meta message or status object, verbatim, for audit
--                and replay.

create table whatsapp_messages (
  id              uuid primary key default gen_random_uuid(),
  conversation_id uuid not null references whatsapp_conversations(id) on delete cascade,
  direction       whatsapp_message_direction not null,
  wamid           text,
  type            whatsapp_message_type not null,
  text_body       text,
  context_wamid   text,
  status          whatsapp_message_status not null,
  error           jsonb,
  raw             jsonb not null,
  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now()
);

alter table whatsapp_messages enable row level security;


-- ----- indexes -----------------------------------------------------------------

-- Contact lookup (webhook upserts, agent worker queries by wa_id).
create index whatsapp_conversations_wa_id_idx on whatsapp_conversations (wa_id);

-- Message history for a given thread (hot path for the agent worker reading
-- context, and for the webhook appending new rows).
create index whatsapp_messages_conversation_id_idx on whatsapp_messages (conversation_id);

-- Idempotency: Meta retries webhook deliveries at-least-once; this collapses
-- duplicate deliveries into a single row. Partial because failed outbound
-- sends have no wamid.
create unique index whatsapp_messages_wamid_idx on whatsapp_messages (wamid) where wamid is not null;
